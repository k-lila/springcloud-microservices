version: "3.9"

networks:
  sales-net:
    name: sales-net
    driver: bridge

services:
  config-server:
    build: ./ConfigServer
    image: images/config-server:latest
    container_name: "config-server"
    networks:
      - sales-net
    ports:
      - 8888:8888

  discovery-server:
    build: ./DiscoveryServer
    image: images/dicovery-server:latest
    container_name: "discovery-server"
    environment:
      CONFIGSERVER_URL: http://config-server:8888
    networks:
      - sales-net
    ports:
      - 9091:9091
    depends_on:
      - config-server

  mongo-database:
    image: mongo:latest
    container_name: "mongo-database"
    networks:
      - sales-net
    ports:
      - 28017:27017

  client-service:
    build: ./ClientService
    image: images/cliente-service:latest
    container_name: "client-service"
    environment:
      PORT: "8080"
      CONFIGSERVER_URL: http://config-server:8888
      MONGODB_URL: mongo-database
      DISCOVERY_HOST: discovery-server
    networks:
      - sales-net
    ports:
      - 8081:8080
    depends_on:
      - config-server
      - discovery-server
      - mongo-database

  product-service:
    build: ./ProductService
    image: images/produto-service:latest
    container_name: "product-service"
    environment:
      PORT: "8080"
      CONFIGSERVER_URL: http://config-server:8888
      MONGODB_URL: mongo-database
      DISCOVERY_HOST: discovery-server
    networks:
      - sales-net
    ports:
      - 8082:8080
    depends_on:
      - config-server
      - discovery-server
      - mongo-database
